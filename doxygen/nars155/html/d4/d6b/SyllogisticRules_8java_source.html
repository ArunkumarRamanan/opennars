224  static void detachment(Sentence mainSentence, Sentence subSentence,
225  int side, Memory memory) {
226  Statement statement = (Statement) mainSentence.getContent();
227  if (!(statement instanceof Implication)
228  && !(statement instanceof Equivalence)) {
229  return;
230  }
231  Term subject = statement.getSubject();
232  Term predicate = statement.getPredicate();
233  Term content;
234  Term term = subSentence.getContent();
235  if ((side == 0) && term.equals(subject)) {
236  content = predicate;
237  } else if ((side == 1) && term.equals(predicate)) {
238  content = subject;
239  } else {
240  return;
241  }
242  if ((content instanceof Statement) && ((Statement) content).invalid()) {
243  return;
244  }
245  Sentence taskSentence = memory.getCurrentTask().getSentence();
246  Sentence beliefSentence = memory.getCurrentBelief();
247  TruthValue beliefTruth = beliefSentence.getTruth();
248  TruthValue truth1 = mainSentence.getTruth();
249  TruthValue truth2 = subSentence.getTruth();
250  TruthValue truth = null;
251  BudgetValue budget;
252  if (taskSentence.isQuestion()) {
253  if (statement instanceof Equivalence) {
254  budget = BudgetFunctions.backward(beliefTruth, memory);
255  } else if (side == 0) {
256  budget = BudgetFunctions.backwardWeak(beliefTruth, memory);
257  } else {
258  budget = BudgetFunctions.backward(beliefTruth, memory);
259  }
260  } else {
261  if (statement instanceof Equivalence) {
262  truth = TruthFunctions.analogy(truth2, truth1);
263  } else if (side == 0) {
264  truth = TruthFunctions.deduction(truth1, truth2);
265  } else {
266  truth = TruthFunctions.abduction(truth2, truth1);
267  }
268  budget = BudgetFunctions.forward(truth, memory);
269  }
