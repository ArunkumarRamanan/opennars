363  private void processNewTask() {
364  Task task;
365  int counter = getNewTasks().size(); // don't include new tasks produced in the current workCycle
366  while (counter-- > 0) {
367  task = getNewTasks().remove(0);
368  if (task.isInput() || (termToConcept(task.getContent()) != null)) { // new input or existing concept
369  immediateProcess(task);
370  } else {
371  Sentence s = task.getSentence();
372  if (s.isJudgment()) {
373  double d = s.getTruth().getExpectation();
374  if (d > Parameters.DEFAULT_CREATION_EXPECTATION) {
375  getNovelTasks().putIn(task); // new concept formation
376  } else {
377  getRecorder().append("!!! Neglected: " + task + "\n");
378  }
379  }
380  }
381  }
382  }

954  public void doTick() {
955  if (DEBUG)
956  if ( running || walkingSteps > 0 || !finishedInputs ) {
957  System.out.println("// doTick: "
958  + "walkingSteps " + walkingSteps
959  + ", clock " + clock
960  + ", getTimer " + getTimer()
961  + "\n// memory.getExportStrings() " + memory.getExportStrings()
962  );
963  System.out.flush();
964  }
965  if (walkingSteps == 0) {
966  boolean reasonerShouldRun = false;
967  for (InputChannel channelIn : inputChannels) {
968  reasonerShouldRun = reasonerShouldRun ||
969  channelIn.nextInput();
970  }
971  finishedInputs = ! reasonerShouldRun;
972  }
973  // forward to output Channels
974  List<String> output = memory.getExportStrings();
975  if (!output.isEmpty()) {
976  for (OutputChannel channelOut : outputChannels) {
977  channelOut.nextOutput(output);
978  }
979  output.clear(); // this will trigger display the current value of timer in Memory.report()
980  }
981  if (running || walkingSteps > 0) {
982  clock++;
983  tickTimer();
984  memory.workCycle(clock);
985  if (walkingSteps > 0) {
986  walkingSteps--;
987  }
988  }
989  }

1000  public void textInputLine(String text) {
1001  if (!text.isEmpty()) {
1002  char c = text.charAt(0);
1003  switch (c) {
1004  case Symbols.RESET_MARK:
1005  reset();
1006  memory.getExportStrings().add(text);
1007  break;
1008  case Symbols.COMMENT_MARK:
1009  return;
1010  default:
1011  // read NARS language or an integer : TODO duplicated code
1012  try {
1013  int i = Integer.parseInt(text);
1014  walk(i);
1015  } catch (NumberFormatException e) {
1016  Task task = StringParser.parseExperience(new StringBuffer(text), memory, clock);
1017  if (task != null) {
1018  memory.inputTask(task);
1019  }
1020  }
1021  break;
1022  }
1023  }
