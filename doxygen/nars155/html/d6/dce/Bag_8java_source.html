154  public boolean putIn(Type newItem) {
155  String newKey = newItem.getKey();
156  Type oldItem = nameTable.put(newKey, newItem);
157  if (oldItem != null) { // merge duplications

187  public Type takeOut() {
188  if (nameTable.isEmpty()) { // empty bag
189  return null;
190  }
191  if (emptyLevel(currentLevel) || (currentCounter == 0)) { // done with the current level
192  currentLevel = DISTRIBUTOR.pick(levelIndex);
193  levelIndex = DISTRIBUTOR.next(levelIndex);
194  while (emptyLevel(currentLevel)) { // look for a non-empty level
195  currentLevel = DISTRIBUTOR.pick(levelIndex);
196  levelIndex = DISTRIBUTOR.next(levelIndex);
197  }
198  if (currentLevel < THRESHOLD) { // for dormant levels, take one item
199  currentCounter = 1;
200  } else { // for active levels, take all current items
201  currentCounter = itemTable.get(currentLevel).size();
202  }
203  }
204  Type selected = takeOutFirst(currentLevel); // take out the first item in the level
205  currentCounter--;
206  nameTable.remove(selected.getKey());
207  refresh();
208  return selected;
209  }

250  private Type intoBase(Type newItem) {
251  Type oldItem = null;
252  int inLevel = getLevel(newItem);
253  if (nameTable.size() > capacity) { // the bag is full
254  int outLevel = 0;
255  while (emptyLevel(outLevel)) {
256  outLevel++;
257  }
258  if (outLevel > inLevel) { // ignore the item and exit
259  return newItem;
260  } else { // remove an old item in the lowest non-empty level
261  oldItem = takeOutFirst(outLevel);
262  }
263  }
264  itemTable.get(inLevel).add(newItem); // FIFO
265  mass += (inLevel + 1); // increase total mass
266  refresh(); // refresh the window
267  return oldItem; // TODO return null is a bad smell
268  }
