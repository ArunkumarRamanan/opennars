145  private static boolean findSubstitute(char type, Term term1, Term term2,
146  HashMap<Term, Term> map1, HashMap<Term, Term> map2) {
147  Term t;
148  if ((term1 instanceof Variable) && ((Variable) term1).getType() == type) {
149  t = map1.get((Variable) term1);
150  if (t != null) {
151  return t.equals(term2);
152  } else {
153  map1.put((Variable) term1, term2);
154  return true;
155  }
156  }
157  if ((term2 instanceof Variable) && ((Variable) term2).getType() == type) {
158  t = map2.get((Variable) term2);
159  if (t != null) {
160  return t.equals(term1);
161  } else {
162  map2.put((Variable) term2, term1);
163  return true;
164  }
165  }
166  if ((term1 instanceof CompoundTerm) && term1.getClass().equals(term2.getClass())) {
167  CompoundTerm cTerm1 = (CompoundTerm) term1;
168  CompoundTerm cTerm2 = (CompoundTerm) term2;
169  if (cTerm1.size() != (cTerm2).size()) {
170  return false;
171  }
172  for (int i = 0; i < cTerm1.size(); i++) {
173  Term t1 = cTerm1.componentAt(i);
174  Term t2 = cTerm2.componentAt(i);
175  boolean haveSub = findSubstitute(type, t1, t2, map1, map2);
176  if (!haveSub) {
177  return false;
178  }
179  }
180  return true;
181  }
182  if (!(term1 instanceof Variable) && !(term2 instanceof Variable) && term1.equals(term2)) {// for constant, also shortcut for variable and compound
183  return true;
184  }
