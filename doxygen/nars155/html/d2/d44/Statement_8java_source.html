191  public static boolean invalidStatement(Term subject, Term predicate) {
192  if (subject.equals(predicate)) {
193  return true;
194  }
195  if ((subject instanceof CompoundTerm) && ((CompoundTerm) subject).containComponent(predicate)) {
196  return true;
197  }
198  if ((predicate instanceof CompoundTerm) && ((CompoundTerm) predicate).containComponent(subject)) {
199  return true;
200  }
201  if ((subject instanceof Statement) && (predicate instanceof Statement)) {
202  Statement s1 = (Statement) subject;
203  Statement s2 = (Statement) predicate;
204  Term t11 = s1.getSubject();
205  Term t12 = s1.getPredicate();
206  Term t21 = s2.getSubject();
207  Term t22 = s2.getPredicate();
208  if (t11.equals(t22) && t12.equals(t21)) {
209  return true;
210  }
211  }
212  return false;
213  }
