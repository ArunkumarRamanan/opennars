68  public static Term make(Term term1, Term term2, Memory memory) {
69  TreeSet<Term> set;
70  if ((term1 instanceof SetInt) && (term2 instanceof SetInt)) {
71  set = new TreeSet<Term>(((CompoundTerm) term1).cloneComponents());
72  set.addAll(((CompoundTerm) term2).cloneComponents()); // set union
73  return SetInt.make(set, memory);
74  }
75  if ((term1 instanceof SetExt) && (term2 instanceof SetExt)) {
76  set = new TreeSet<Term>(((CompoundTerm) term1).cloneComponents());
77  set.retainAll(((CompoundTerm) term2).cloneComponents()); // set intersection
78  return SetExt.make(set, memory);
79  }
80  if (term1 instanceof IntersectionExt) {
81  set = new TreeSet<Term>(((CompoundTerm) term1).cloneComponents());
82  if (term2 instanceof IntersectionExt) {

84  } // (&,(&,P,Q),(&,R,S)) = (&,P,Q,R,S)

87  } // (&,(&,P,Q),R) = (&,P,Q,R)
88  } else if (term2 instanceof IntersectionExt) {

91  } else {
92  set = new TreeSet<Term>();
93  set.add((Term) term1.clone());
94  set.add((Term) term2.clone());
95  }
