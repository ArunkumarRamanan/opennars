149  public static Term make(String op, ArrayList<Term> arg, Memory memory) {
150  if (op.length() == 1) {
151  if (op.charAt(0) == Symbols.SET_EXT_OPENER) {
152  return SetExt.make(arg, memory);
153  }
154  if (op.charAt(0) == Symbols.SET_INT_OPENER) {
155  return SetInt.make(arg, memory);
156  }
157  if (op.equals(Symbols.INTERSECTION_EXT_OPERATOR)) {
158  return IntersectionExt.make(arg, memory);
159  }
160  if (op.equals(Symbols.INTERSECTION_INT_OPERATOR)) {
161  return IntersectionInt.make(arg, memory);
162  }
163  if (op.equals(Symbols.DIFFERENCE_EXT_OPERATOR)) {
164  return DifferenceExt.make(arg, memory);
165  }
166  if (op.equals(Symbols.DIFFERENCE_INT_OPERATOR)) {
167  return DifferenceInt.make(arg, memory);
168  }
169  if (op.equals(Symbols.PRODUCT_OPERATOR)) {
170  return Product.make(arg, memory);
171  }
172  if (op.equals(Symbols.IMAGE_EXT_OPERATOR)) {
173  return ImageExt.make(arg, memory);
174  }
175  if (op.equals(Symbols.IMAGE_INT_OPERATOR)) {
176  return ImageInt.make(arg, memory);
177  }
178  }
179  if (op.length() == 2) {
180  if (op.equals(Symbols.NEGATION_OPERATOR)) {
181  return Negation.make(arg, memory);
182  }
183  if (op.equals(Symbols.DISJUNCTION_OPERATOR)) {
184  return Disjunction.make(arg, memory);
185  }
186  if (op.equals(Symbols.CONJUNCTION_OPERATOR)) {
187  return Conjunction.make(arg, memory);
188  }
189  }
190  return null;

518  if (containVar()) {
519  for (int i = 0; i < components.size(); i++) {
520  Term term = componentAt(i);
521  if (term instanceof Variable) {
522  Variable var;
523  if (term.getName().length() == 1) { // anonymous variable from input
524  var = new Variable(term.getName().charAt(0) + "" + (map.size() + 1));

588  private void prepareComponentLinks(ArrayList<TermLink> componentLinks, short type, CompoundTerm term) {
589  Term t1, t2, t3; // components at different levels
590  for (int i = 0; i < term.size(); i++) { // first level components
591  t1 = term.componentAt(i);
592  if (t1.isConstant()) {
593  componentLinks.add(new TermLink(t1, type, i));
594  }
595  if ((t1 instanceof Conjunction) && ((this instanceof Equivalence) || ((this instanceof Implication) && (i == 0)))) {
596  ((CompoundTerm) t1).prepareComponentLinks(componentLinks, TermLink.COMPOUND_CONDITION, (CompoundTerm) t1);
597  } else if (t1 instanceof CompoundTerm) {
598  for (int j = 0; j < ((CompoundTerm) t1).size(); j++) { // second level components
599  t2 = ((CompoundTerm) t1).componentAt(j);
600  if (t2.isConstant()) {
601  if ((t1 instanceof Product) || (t1 instanceof ImageExt) || (t1 instanceof ImageInt)) {
602  if (type == TermLink.COMPOUND_CONDITION) {
603  componentLinks.add(new TermLink(t2, TermLink.TRANSFORM, 0, i, j));
604  } else {
605  componentLinks.add(new TermLink(t2, TermLink.TRANSFORM, i, j));
606  }
607  } else {
608  componentLinks.add(new TermLink(t2, type, i, j));
609  }
610  }
611  if ((t2 instanceof Product) || (t2 instanceof ImageExt) || (t2 instanceof ImageInt)) {
612  for (int k = 0; k < ((CompoundTerm) t2).size(); k++) {
613  t3 = ((CompoundTerm) t2).componentAt(k);
614  if (t3.isConstant()) { // third level
615  if (type == TermLink.COMPOUND_CONDITION) {
616  componentLinks.add(new TermLink(t3, TermLink.TRANSFORM, 0, i, j, k));
617  } else {
618  componentLinks.add(new TermLink(t3, TermLink.TRANSFORM, i, j, k));
619  }
620  }
621  }
622  }
623  }
