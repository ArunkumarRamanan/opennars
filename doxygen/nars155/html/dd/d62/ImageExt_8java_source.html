103  public static Term make(Product product, Term relation, short index, Memory memory) {
104  if (relation instanceof Product) {
105  Product p2 = (Product) relation;
106  if ((product.size() == 2) && (p2.size() == 2)) {
107  if ((index == 0) && product.componentAt(1).equals(p2.componentAt(1))) { // (/,_,(*,a,b),b) is reduced to a
108  return p2.componentAt(0);
109  }
110  if ((index == 1) && product.componentAt(0).equals(p2.componentAt(0))) { // (/,(*,a,b),a,_) is reduced to b
111  return p2.componentAt(1);
112  }
113  }
114  }
115  ArrayList<Term> argument = product.cloneComponents();
116  argument.set(index, relation);
117  return make(argument, index, memory);
