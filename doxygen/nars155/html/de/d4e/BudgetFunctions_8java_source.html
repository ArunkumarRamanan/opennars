118  static BudgetValue revise(TruthValue tTruth, TruthValue bTruth,
119  TruthValue truth, boolean feedbackToLinks, Memory memory) {
120  float difT = truth.getExpDifAbs(tTruth);
121  Task task = memory.getCurrentTask();
122  task.decPriority(1 - difT);
123  task.decDurability(1 - difT);
124  if (feedbackToLinks) {
125  TaskLink tLink = memory.getCurrentTaskLink();
126  tLink.decPriority(1 - difT);
127  tLink.decDurability(1 - difT);
128  TermLink bLink = memory.getCurrentBeliefLink();
129  float difB = truth.getExpDifAbs(bTruth);
130  bLink.decPriority(1 - difB);
131  bLink.decDurability(1 - difB);
132  }
133  float dif = truth.getConfidence()
134  - Math.max(tTruth.getConfidence(), bTruth.getConfidence());
135  float priority = or(dif, task.getPriority());
136  float durability = or(dif, task.getDurability());
137  float quality = truthToQuality(truth);
138  return new BudgetValue(priority, durability, quality);
139  }

333  private static BudgetValue budgetInference(float qual, int complexity,
334  Memory memory) {
335  Item t = memory.getCurrentTaskLink();
336  if (t == null)
337  t = memory.getCurrentTask();
338  float priority = t.getPriority();
339  float durability = t.getDurability();
340  float quality = (float) (qual / Math.sqrt(complexity));
341  TermLink bLink = memory.getCurrentBeliefLink();
342  if (bLink != null) {
343  priority = aveAri(priority, bLink.getPriority());
344  durability = aveAri(durability, bLink.getDurability());
345  bLink.incPriority(quality);
346  }
347  return new BudgetValue(and(priority, quality),
348  and(durability, quality), quality);
349  }
