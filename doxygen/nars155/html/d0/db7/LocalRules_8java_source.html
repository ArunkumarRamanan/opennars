217  private static void inferToSym(Sentence judgment1, Sentence judgment2,
218  Memory memory) {
219  Statement s1 = (Statement) judgment1.getContent();
220  Term t1 = s1.getSubject();

222  Term content;

229  TruthValue value2 = judgment2.getTruth();

306  private static void convertedJudgment(TruthValue newTruth,
307  BudgetValue newBudget, Memory memory) {
308  Statement content = (Statement) memory.getCurrentTask().getContent();
309  Statement beliefContent = (Statement) memory.getCurrentBelief()
310  .getContent();

312  Term predT = content.getPredicate();

314  Term predB = beliefContent.getPredicate();
315  Term otherTerm;
316  if (Variable.containVarQuery(subjT.getName())) {
317  otherTerm = (predT.equals(subjB)) ? predB : subjB;
318  content = Statement.make(content, otherTerm, predT, memory);
319  }
320  if (Variable.containVarQuery(predT.getName())) {
321  otherTerm = (subjT.equals(subjB)) ? predB : subjB;
322  content = Statement.make(content, subjT, otherTerm, memory);
323  }
324  memory.singlePremiseTask(content, Symbols.JUDGMENT_MARK, newTruth,
325  newBudget);
