85  static TruthValue revision(TruthValue v1, TruthValue v2) {
86  float f1 = v1.getFrequency();
87  float f2 = v2.getFrequency();
88  float c1 = v1.getConfidence();
89  float c2 = v2.getConfidence();
90  float w1 = c2w(c1);
91  float w2 = c2w(c2);
92  float w = w1 + w2;
93  float f = (w1 * f1 + w2 * f2) / w;
94  float c = w2c(w);
95  return new TruthValue(f, c);
