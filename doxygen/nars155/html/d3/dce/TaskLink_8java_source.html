86  Term bTerm = termLink.getTarget();

91  int next = 0;

95  if (linkKey.equals(recordedLinks[next])) {
96  if (currentTime < recordingTime[next] + Memory.Parameters.TERM_LINK_RECORD_LENGTH) {
97  return false;
98  } else {

100  return true;

102  }
103  }

105  recordedLinks[next] = linkKey; // add knowledge reference to recordedLinks
106  recordingTime[next] = currentTime;
