149  private static String getTruthString(StringBuffer s) throws InvalidInputException {
150  int last = s.length() - 1;
151  if (s.charAt(last) != TRUTH_VALUE_MARK) { // use default
152  return null;
153  }
154  int first = s.indexOf(TRUTH_VALUE_MARK + ""); // looking for the beginning
155  if (first == last) { // no matching closer
156  throw new InvalidInputException("missing truth mark");
157  }
158  String truthString = s.substring(first + 1, last).trim();
159  if (truthString.length() == 0) { // empty usage
160  throw new InvalidInputException("empty truth");
161  }
162  s.delete(first, last + 1); // remaining input to be processed outside
163  s.trimToSize();
164  return truthString;
165  }

200  private static BudgetValue parseBudget(String s, char punctuation, TruthValue truth) throws InvalidInputException {
201  float priority, durability;

213  }
214  if (s != null) { // overrite default
215  int i = s.indexOf(VALUE_SEPARATOR);
216  if (i < 0) { // default durability
217  priority = Float.parseFloat(s);
218  } else {
219  priority = Float.parseFloat(s.substring(0, i));
220  durability = Float.parseFloat(s.substring(i + 1));
221  }
222  }
223  float quality = (truth == null) ? 1 : BudgetFunctions.truthToQuality(truth);
224  return new BudgetValue(priority, durability, quality);

241  public static Term parseTerm(String s0, Memory memory) {
242  String s = s0.trim();
243  try {
244  if (s.length() == 0) {
245  throw new InvalidInputException("missing content");
246  }
247  Term t = memory.nameToListedTerm(s); // existing constant or operator
248  if (t != null) {
249  return t;
250  } // existing Term
251  int index = s.length() - 1;
252  char first = s.charAt(0);
253  char last = s.charAt(index);
254  switch (first) {
255  case COMPOUND_TERM_OPENER:
256  if (last == COMPOUND_TERM_CLOSER) {
257  return parseCompoundTerm(s.substring(1, index), memory);
258  } else {
259  throw new InvalidInputException("missing CompoundTerm closer");
260  }
261  case SET_EXT_OPENER:
262  if (last == SET_EXT_CLOSER) {
263  return SetExt.make(parseArguments(s.substring(1, index) + ARGUMENT_SEPARATOR, memory), memory);
264  } else {
265  throw new InvalidInputException("missing ExtensionSet closer");
266  }
267  case SET_INT_OPENER:
268  if (last == SET_INT_CLOSER) {
269  return SetInt.make(parseArguments(s.substring(1, index) + ARGUMENT_SEPARATOR, memory), memory);
270  } else {
271  throw new InvalidInputException("missing IntensionSet closer");
272  }
273  case STATEMENT_OPENER:
274  if (last == STATEMENT_CLOSER) {
275  return parseStatement(s.substring(1, index), memory);
276  } else {
277  throw new InvalidInputException("missing Statement closer");
278  }
279  default:
280  return parseAtomicTerm(s);
281  }
282  } catch (InvalidInputException e) {
283  String message = " !!! INVALID INPUT: parseTerm: " + s + " --- " + e.getMessage();
284  System.out.println(message);
285  showWarning(message);
286  }
287  return null;
288  }

368  private static ArrayList<Term> parseArguments(String s0, Memory memory) throws InvalidInputException {
369  String s = s0.trim();
370  ArrayList<Term> list = new ArrayList<Term>();
371  int start = 0;
372  int end = 0;
373  Term t;
374  while (end < s.length() - 1) {
375  end = nextSeparator(s, start);
376  t = parseTerm(s.substring(start, end), memory); // recursive call
377  list.add(t);
378  start = end + 1;
379  }
380  if (list.isEmpty()) {
381  throw new InvalidInputException("null argument");
382  }
383  return list;
384  }

393  private static int nextSeparator(String s, int first) {
394  int levelCounter = 0;
395  int i = first;
396  while (i < s.length() - 1) {
397  if (isOpener(s, i)) {
398  levelCounter++;
399  } else if (isCloser(s, i)) {
400  levelCounter--;
401  } else if (s.charAt(i) == ARGUMENT_SEPARATOR) {
402  if (levelCounter == 0) {
403  break;
404  }
405  }
406  i++;
407  }
408  return i;
409  }

416  private static int topRelation(String s) { // need efficiency improvement
417  int levelCounter = 0;
